```python
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
import numpy as np
from scipy import stats

def sample_size_for_proportion(baseline, mde, alpha=0.05, power=0.80):
    """Two-sided z-test for difference in proportions."""
    p1 = baseline
    p2 = baseline + mde
    z_alpha = stats.norm.ppf(1 - alpha / 2)
    z_beta = stats.norm.ppf(power)
    pooled_p = (p1 + p2) / 2
    n = ((z_alpha * np.sqrt(2 * pooled_p * (1 - pooled_p)) +
          z_beta * np.sqrt(p1 * (1 - p1) + p2 * (1 - p2))) ** 2) / (mde ** 2)
    return int(np.ceil(n))

# Experiment parameters — replace with real values from your analysis
baseline_rate = 0.03   # e.g. 3% CTR baseline for digital ads
mde = 0.006            # minimum detectable effect (absolute)
alpha = 0.05

chosen_n = sample_size_for_proportion(baseline_rate, mde, alpha, 0.80)
print(f"Required sample size per arm: {chosen_n:d}")
print(f"Total sample size (2 arms): {chosen_n * 2:d}")

# Power curve
sample_sizes = np.arange(100, min(chosen_n * 3, 50000), 100)
powers = []
for n in sample_sizes:
    se = np.sqrt(2 * baseline_rate * (1 - baseline_rate) / n)
    z = mde / se - stats.norm.ppf(1 - alpha / 2)
    powers.append(float(stats.norm.cdf(z)))

fig, ax = plt.subplots(figsize=(8, 5))
ax.plot(sample_sizes, powers, color='steelblue', linewidth=2)
ax.axhline(0.80, color='gray', linestyle='--', label='80% power target')
ax.axvline(chosen_n, color='crimson', linestyle='--', label=f'Required n={chosen_n:d}')
ax.set_xlabel('Sample Size per Arm')
ax.set_ylabel('Statistical Power')
ax.set_title('Power Curve: Top Priority Experiment')
ax.legend()
ax.set_ylim(0, 1)
plt.tight_layout()
# Do NOT call plt.close() or plt.savefig() — pipeline captures open figures automatically.
```
