## Python Power Analysis Template
Use scipy.stats for power calculations:
```python
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
import numpy as np
from scipy import stats
import io, base64

# Example: proportion test (e.g., CTR experiment)
# baseline_rate = 0.03  # control CTR
# mde = 0.005           # minimum detectable effect (absolute)
# alpha = 0.05
# power = 0.80

def sample_size_for_proportion(baseline, mde, alpha=0.05, power=0.80):
    """Two-sided z-test for difference in proportions."""
    p1 = baseline
    p2 = baseline + mde
    z_alpha = stats.norm.ppf(1 - alpha / 2)
    z_beta = stats.norm.ppf(power)
    pooled_p = (p1 + p2) / 2
    n = ((z_alpha * np.sqrt(2 * pooled_p * (1 - pooled_p)) +
          z_beta * np.sqrt(p1 * (1 - p1) + p2 * (1 - p2))) ** 2) / (mde ** 2)
    return int(np.ceil(n))

# Plot power curve
sample_sizes = np.arange(100, 10000, 50)
powers = []
for n in sample_sizes:
    # compute power at this n — use normal approximation
    se = np.sqrt(2 * baseline_rate * (1 - baseline_rate) / n)
    z = mde / se - stats.norm.ppf(1 - alpha / 2)
    powers.append(stats.norm.cdf(z))

fig, ax = plt.subplots(figsize=(8, 5))
ax.plot(sample_sizes, powers, color='steelblue', linewidth=2)
ax.axhline(0.80, color='gray', linestyle='--', label='80% power target')
chosen_n = sample_size_for_proportion(baseline_rate, mde, alpha, 0.80)
ax.axvline(chosen_n, color='crimson', linestyle='--', label='Required n=' + str(chosen_n))
ax.set_xlabel('Sample Size per Arm')
ax.set_ylabel('Statistical Power')
ax.set_title('Power Curve: [Experiment Name]')
ax.legend()
ax.set_ylim(0, 1)

buf = io.BytesIO()
plt.savefig(buf, format='png', bbox_inches='tight', dpi=150)
plt.close()
buf.seek(0)
b64 = base64.b64encode(buf.read()).decode('utf-8')
print('CHART_BASE64:' + b64)
```

## Experiment Guardrails
- Never run more than one variable per experiment (isolate the hypothesis).
- Do not set MDE below 1% absolute for conversion metrics — this leads to underpowered tests.
- Flag experiments that require >8 weeks to reach significance given estimated daily traffic.
- Do not test pricing and creative simultaneously in the same experiment.
